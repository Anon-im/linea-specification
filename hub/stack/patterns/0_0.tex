We let $\zeroZeroSP_{i}$ stand for the following collection of constraints whereby all four stack items are empty and the stack height remains unadulterated:
% \begin{description}
% 	\item[\underline{Stack Item $n^\circ\,1, 2, 3, 4$:}] $\emptySP_{i}$; 
% 	\item[\underline{Height update:}]  $\stackHeight{}\new{}_{i} = \stackHeight{}_{i}$,
% \end{description}
\[
	\zeroZeroSP_{i}
	\iff
	\left\{
	\begin{array}{l}
		\emptySP_{i}									\\
		\stackHeight{}\new{}_{i} = \stackHeight{}_{i}	\\
	\end{array}
	\right.
\]
