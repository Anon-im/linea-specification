We impose the following constraints.
\begin{description}
	\item[\underline{Forward setting of \emph{some} \inst{CREATE}-scenario:}] we enforce that
		\[
			\If
			\left\{ \begin{array}{lcl}
				\peekStack_{i}	       & = & 1 \\
				\stackDecCreateFlag_{i}     & = & 1 \\
				\ct_{i}                & = & 0 \\
				\stackSux_{i}          & = & 0 \\
			\end{array} \right.
			\Then
			\left\{ \begin{array}{lcl}
				\peekScenario_{i + 2}                 & = & 1 \\
				\scenCreateSum_{i + 2}                & = & 1 \\
			\end{array} \right.
		\]
\end{description}
