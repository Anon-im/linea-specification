The heartbeat is very simple:
\begin{enumerate}
	\item $\iomf_{0} = 0$
	\item \If $\iomf_{i} = 0$ \Then:
	\begin{enumerate}
		\item $\inRange _{i} = 0$
		\item $\blockNumber\high_{i}=0$ 
		\item $\blockNumber\low_{i}=0$  
	\end{enumerate}
	\item \If $\iomf_{i} \neq 0$ \Then:
	\begin{enumerate}
		\item $\iomf _{i} = 1$
		\item $\iomf_{i+1} = \iomf_{i}$
	\end{enumerate}
\end{enumerate}
