\noindent As usual, we define a column $\col{X}$ to be \textbf{counter-constant} if it satisfies
\[
	\ct_{i} \neq 0 \implies \col{X}_{i} = \col{X}_{i - 1}.
\]
The following columns are required to be counter-constant:
\begin{multicols}{4}
\begin{enumerate}
	\item $\argOneHi$
	\item $\argOneLo$
	\item $\argTwoHi$
	\item $\argTwoLo$
	\item $\resHi$
	\item $\resLo$
	\item $\INST$
	\item $\maxCt$
	\item $\pivotByte$
	\item $\bit{b_4}$
	\item $\lowFour$
	\item $\NEG$
	\item $\smallness$
\end{enumerate}
\end{multicols}
