	\item[\underline{Selector:}] $\done_{i}=1$
	\item[\underline{Source columns:}] ---
		\begin{multicols}{3}
			\begin{enumerate}
				\item $0$
				\item $\remainder_{i}$
				\item $0$
				\item $\divisor_{i}$
				\item $1$
				\item $\inst{LT}$
			\end{enumerate}
		\end{multicols}
	\item[\underline{Target columns:}] ---
		\begin{multicols}{3}
		\begin{enumerate}
			\item $\argOneHi$
			\item $\argOneLo$
			\item $\argTwoHi$
			\item $\argTwoLo$

\end{description}
