	\item[Selector:] none required;
	\item[Source columns:] we use the following:
		\begin{multicols}{2}
			\begin{enumerate}
				\item \INST{}
				\item $\decWordCost$
				\item $\decByteCost$
				\item[\vspace{\fill}]
				\item[\vspace{\fill}]
				\item $\decMxpType{1}$
				\item $\decMxpType{2}$
				\item $\decMxpType{3}$
				\item $\decMxpType{4}$
				\item $\decMxpType{5}$
					% \item[\vspace{\fill}]
			\end{enumerate}
		\end{multicols}
	\item[Target columns:] we use the following:
		\begin{multicols}{2}
			\begin{enumerate}
				\item \INST{}
				\item $\wordCost$
				\item $\byteCost$
				\item[\vspace{\fill}]
				\item[\vspace{\fill}]
				\item $\mxpType{1}$
				\item $\mxpType{2}$

		\end{multicols}
