				\item $\totalSize_{i}$
%				\item $\successBit_{i}$
			\end{enumerate}
		\end{multicols}
	\item[\underline{Target columns:}] from the \logDataMod{} module: 
		\begin{multicols}{3}
			\begin{enumerate}
				\item $\absLogNum_{j}$
				\item $\index_{j}$
				\item $\limb_{j}$
				\item $\limbSize_{j}$

			\end{enumerate}
		\end{multicols}
\end{description}
